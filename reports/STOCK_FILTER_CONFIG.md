# 过滤创业板 (300/301 开头) 和科创板 (688 开头)
def is_valid_stock(code):
    code_str = str(code)
    # 排除创业板 (300/301) 和科创板 (688)
    if code_str.startswith('300') or code_str.startswith('301'):
        return False
    if code_str.startswith('688'):
        return False
    return True

df = df[df['代码'].apply(is_valid_stock)].copy()
